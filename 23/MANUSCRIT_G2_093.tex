\begin{rubrica}

Moss\textit{en} auzias march\footnote{\textit{Al marge dret}: C. 11}

\end{rubrica}

\begin{estrofa}

\numero{1} Leixant apart / lestil dels trobadors

\numero{2} qui per escalf / trespasen veritat

\numero{3} esostraent / mon voler affectat

\numero{4} perque nom torb / direlque trop en uos

\numero{5} Tot monparlar / als qui nous auranvista

\numero{6} res no valdra / carfe noy donaran

\numero{7} elos vehents / que dins vos no veuran

\numero{8} enveura mi / lur arma ss\textit{er}a trista

\end{estrofa}



\begin{estrofa}

\numero{9} [133r] Lull del hom pech / no ha tan foscavista

\numero{10} que vostre cors / no jutge per gentil

\numero{11} nol coneix tal / com lo qui essubtil

\numero{12} hoc lacolor / mas no sap delallista

\numero{13} Quant es delcors / menys departicipar

\numero{14} ab lesperit / coneix be lo grosser

\numero{15} vostracolor / yeltall pot besaber

\numero{16} masja del gesch / no porabeparlar

\end{estrofa}



\begin{estrofa}

\numero{17} Tots som grossers / enpoder explicar

\numero{18} lo que mereix / huncors bell / e honest

\numero{19} jouens gentils / bons sabents lan request

\numero{20} efamejants / los couench endurar

\numero{21} Lo vostre seny / faso qualtrenobasta

\numero{22} que sap regir la molta subtilesa

\numero{23} enfertotbe sadorm envos peresa

\numero{24} verge no sou / per que deu ne volch casta

\end{estrofa}



\begin{estrofa}

\numero{25} Sol per auos basta / la bonapasta

\numero{26} que deu retench / perfer singulars dones

\numero{27} fetes nassats / molt sauies ebones

\numero{28} mascomplime\textit{n}t / dona terresal tasta

\numero{29} hauent ensi tangran / coneximent

\numero{30} queres nol fall / que tota nos conega

\numero{31} alhom deuot sabellesa encega

\numero{32} past dentenents / esson entenime\textit{n}t

\end{estrofa}



\begin{estrofa}

\numero{34} tanpasiffich / com v\textit{ost}re seny regeix

\numero{35} subtilitats / quelentendre\textit{us} nodrex\footnote{\textit{Mot amb un sufix
esborrat, il$\cdot{}$legible}.}

\numero{36} edelcos bell / senscolpal mouiment

\numero{37} Tangran delit / tot hom entenent ha

\numero{38} eocupat / setroban uos entendre

\numero{39} quelo desig del cors / nos potestendre

\numero{40} aleig voler / ans com amort esta

\end{estrofa}
