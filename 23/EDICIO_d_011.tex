\poema{(d 11) [23]}\\

\begin{rubrica}

\textparagraph\  DE AMOR CANTO. XI.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} L\textsc{e}ixant apart / l'estil dels trobad\'{o}s

\numero{2} qui per escalf / traspassen veritat,

\numero{3} e sostrahent / mon voler affectat

\numero{4} perque nom trob / dir el que trob en vos.

\numero{5} Tot  mon parlar / als que no vs hauran vista

\numero{6} res noy valra / car fe noy donaran

\numero{7} e los vehents / que dins vos no veuran

\numero{8} en creure'mi / llur arma sera trista.

\end{estrofa}



\begin{estrofa}

\numero{9} [17v] Llull de hom pech / no ha tant fosca vista

\numero{10} que vostre cos / no jutje per gentil:

\numero{11} nol coneix tal / com lo qui es suptil:

\numero{12} hoc la color / mas no sab dela lista.

\numero{13} Quant es del cos / menys de parti\c{c}ipar

\numero{14} del espirit / coneix be lo grosser:

\numero{15} vostre color / y el tall pot be saber

\numero{16} mas ja del gest / no pora ben parlar.

\end{estrofa}



\begin{estrofa}

\numero{17} \textparagraph\  Tots som grossers / en poder esplicar

\numero{18} \c{c}o que mereix / vn bell cos e honest,

\numero{19} iouens gentils / bens sabents l'an request:

\numero{20} e famejants / los conuench en durar.

\numero{21} Lo vostre seny / que altre noy b\'{a}sta

\numero{22} que sab regir / la molta sobtilesa

\numero{23} en fer tot be / s'adorm en vos peresa

\numero{24} verge no sou / perche Deu ne volch c\'{a}sta.

\end{estrofa}



\begin{estrofa}

\numero{25} \textparagraph\  Sol pera vos / basta la bona pasta

\numero{26} que Deu retench / per fer singulars dones

\numero{27} fets neha sats / molt sauies e bones:

\numero{28} mas compliment / dona Teresa'l t\'{a}sta.

\numero{29} Hauent ensi / tan gran coneiximent

\numero{30} que res nol fall / que tota nos'conega

\numero{31} al hom de vot /sa bellesa en \c{c}ega:

\numero{32} past de'ntents /es son enteniment.

\end{estrofa}



\begin{estrofa}

\numero{33} [18r] \textparagraph\  Vene\c{c}ians / no han lo regiment

\numero{34} tan pa\c{c}ifich /com vostre seny regeix

\numero{35} subtilitats / quel entendrens nodreix

\numero{36} e del cos bell / sens culpa'l mouiment.

\numero{37} Tan gran delit / tot hom entenent ha

\numero{38} e ocupat / se troba'n vos entendre

\numero{39} que lo desig / del cos no pot estendre

\numero{40} a leig voler / ans com amort est\'{a}.

\end{estrofa}

\numero{42} tant queus pogues / fer corona inuisible

\numero{43} meriu  la vos / car la qui es visible
