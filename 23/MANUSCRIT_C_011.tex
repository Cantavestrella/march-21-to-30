\begin{estrofa}

\numero{1} Leixant a part / l'estil dels trobados

\numero{2} qui per scalf / trespassen ueritat

\numero{3} e sostrahent / mon uoler affectat

\numero{4} perque nom trob / dire'lque trobe en uos

\numero{5} tot mon parlar / als que nous hauran uista

\numero{6} res noy ualra / car fe noy donaran

\numero{7} elos uehens / que dins uos noueuran

\numero{8} en creure mi / lur arma sera trista.

\end{estrofa}



\begin{estrofa}

\numero{9} Lull del hom pech / no ha tan fosca uista

\numero{10} que uostre cos / no iutie per gentill

\numero{11} nolconeix tal / com lo qui es suptill

\numero{12} hoc la color / mas no sab dela lista

\numero{13} quant es del cos / menys participar

\numero{14} del'esperit, / coneix be lo grosser;

\numero{15} uostre color / il tall pot be saber;

\numero{16} mas ia del gest / no pora ben parlar.

\end{estrofa}



\begin{estrofa}

\numero{17} [19v] Tots son grosses / en poder explicar

\numero{18} \c{c}o que mereix / un bel cos, e\sout{t} honest.

\numero{19} jouens gentills / ben sabents l'han request

\numero{20} e fameiants / los conuench endurar.

\numero{21} louostre seny / fa \c{c}o qu'altre no basta

\numero{22} que sab regir / la molta soptilesa.

\numero{23} enfer tot be / s'adorm en uos per\'{e}sa.

\numero{24} uerge no sou / perque Deu no uolch, casta,

\end{estrofa}



\begin{estrofa}

\numero{25} Sol per auos / bast\`{a} la bona pasta

\numero{26} que deu retench / per fer singulars dones.

\numero{27} fetes n'hassats / molt sauies e bones

\numero{28} mas compliment / Dona Teresal tasta.

\numero{29} hauent ensi / tan gran coneiximent,

\numero{30} que res nol fall / que tota nos con\'{e}ga.

\numero{31} al hom deuot / sa bellesa enc\'{e}ga.

\numero{32} past d'entents / es son enteniment.

\end{estrofa}



\begin{estrofa}

\numero{33} Vene\c{c}ians / no han lo regiment

\numero{34} tan pa\c{c}ifich / com uostre seny regeix

\numero{35} subti\interlin{li}tats / que l'entendrens nodeix.

\numero{36} e del cos bell / sens culpal'\sout{el} mouiment;

\numero{37} [20r] tan gran delit / tot hom entenent ha,

\numero{38} e occupat / se trob'enuos entendre,

\numero{39} que lo desig / del cos no pot estendre

\numero{40} a leig uoler; / ans com a mort'sta.

\end{estrofa}
