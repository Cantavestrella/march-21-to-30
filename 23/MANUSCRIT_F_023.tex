\begin{rubrica}

Altra

\end{rubrica}

\begin{estrofa}

\numero{1} Lexant apart lestil dels trobado\sout{r}s

\numero{2} Quj per escalf trespassen ueritat

\numero{3} E sostrahent mon uoler affectat

\numero{4} Per que nom t\sout{r}o\interlin{r}b direl que trob enuos

\numero{5} Tot mon parlar als quj no\textit{us} hauran vista

\numero{6} Res no ualrra car fe noy donaran

\numero{7} E los vehents que dins uos no veuran

\numero{8} En creuramj llur arma sera trista

\end{estrofa}



\begin{estrofa}

\numero{9} [23v] Lull del hom pech no ha tan fosqua vista

\numero{10} Que uostre cors no jutge per gentil

\numero{11} Nol coneix tal com lo quj es suptil

\numero{12} Hoc la calor mas no sap d\textit{e}la lista

\numero{13} Quant es del cors menys de participar

\numero{14} Ab lesperit coneix be lo grosser

\numero{15} Vostra color yell tall pot be saber

\numero{16} Mas ga del gest no pora be parlar

\end{estrofa}



\begin{estrofa}

\numero{17} Tots som grossers en poder explicar

\numero{18} Co que mereix hun bell cors e honest

\numero{19} Jouens gentils bons sabents lan request

\numero{20} E fameiants los coue endurar

\numero{21} Lo uostre seny fa co qaltre no basta

\numero{22} Que sab regir la molta subtilea

\numero{23} En fer tot be sadorm en uos perea

\numero{24} Verge no sou per que deu ne uolch casta

\end{estrofa}



\begin{estrofa}

\numero{25} Sol per auos basta la bona pasta

\numero{26} Que deu retench per fer singulars dones

\numero{27} Ffetes nassats molt saujes e bones

\numero{28} Mas compliment dona teresal tasta\footnote{\textit{Al marge esquerre, en roig i
casi il$\cdot{}$legible, d'una altra m\`{a}: }Dona Teresa.}

\numero{29} Hauent en si tan gran coneximent

\numero{30} Que res nol fall que tota nos conega

\numero{31} Al hom deuot sa bellesa encega

\numero{32} Past dentenents es son entenjment

\end{estrofa}



\begin{estrofa}

\numero{33} [24r] Venecians no han lo regiment

\numero{34} Tan pasciffich com uostre seny regeix

\numero{35} Suptilitats quel entendreus nodreix

\numero{36} E del cors bell sens colpal moujment

\numero{37} Tan gran delit tot hom entenent ha

\numero{38} E occupat se troban uos entendre

\numero{39} Que lo desig del cors nos pot estendre

\numero{40} A leig uoler ans com amort esta

\end{estrofa}



\begin{estrofa}

\numero{41} Lir entre carts lo meu poder no fa

\numero{42} Tant que pogues fer coronan uisible

\numero{43} Merju la uos car la quj es visible

\numero{44} Nos deu posar llaon mjraclesta

\end{estrofa}
