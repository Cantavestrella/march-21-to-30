\begin{rubrica}

DE AMOR CANT. XI.

\end{rubrica}

\begin{estrofa}

\numero{1} [11v] L\textsc{e}ixant a part / l'estil dels trobad\'{o}s

\numero{2} qui per escalf / transpassen veritat,

\numero{3} e sostrahent / mon voler afectat

\numero{4} per que nom trob / dir'el que trob en vos.

\numero{5} Tot mon parlar / als qui nous auran vista

\numero{6} res noy valra / car fe noy donaran

\numero{7} e los vehents / que dins vos no veuran

\numero{8} en creur'ami / lur arma sera trista.

\end{estrofa}



\begin{estrofa}

\numero{9} Lull de hom pech / no ha tant fosca vista

\numero{10} que vostre cos / no jutje per gentil,

\numero{11} nel coneix tal / com lo qui es suptil,

\numero{12} hoc la color / mas no sab dela lista.

\numero{13} Quant es del cos / menys de participar

\numero{14} delesperit / coneix be lo grosser:

\numero{15} vostre color / y el tall pot be saber

\numero{16} mas ja del gest / no pora ben parlar.

\end{estrofa}



\begin{estrofa}

\numero{17} Tots som grossers / en poder esplicar

\numero{18} \c{c}o que mereix / vn bell cos e honest,

\numero{19} jouens gentils / ben sabents l'an request:

\numero{20} e famejants / los conuench endurar.

\numero{21} Lo vostre seny / fa \c{c}o qu'altre noy b\'{a}sta

\numero{22} que sab regir / la molta sobtilesa

\numero{23} en fer tot be / s'adorm en vos peresa

\numero{24} verge no sou / per que Deu ne volch c\'{a}sta.

\end{estrofa}



\begin{estrofa}

\numero{25} [12r] Sol pera vos / basta la bona pasta

\numero{26} que Deu retench / per fer singulars dones

\numero{27} fets ne hassats / molt sauies e bones:

\numero{28} mas compliment / dona Teresa'l t\'{a}sta.

\numero{29} Hauent en si / tan gran coneiximent

\numero{30} que res nol fall / que tota nos'conega

\numero{31} al hom deuot / sa bellesa encega:

\numero{32} past d'entenents / es son enteniment.

\end{estrofa}



\begin{estrofa}

\numero{33} Venecians / no han lo regiment

\numero{34} tan pacifich / com vostre seny regeix,

\numero{35} subtilitats / quel entendrens nodreix

\numero{36} e del cos bell / sens culp'al mouiment.

\numero{37} Tan gran delit / tot hom entenent ha

\numero{38} e ocupat / se troba'n vos entendre

\numero{39} que lo desig / del cos nos pot estendre

\numero{40} a leig voler / ans com amort est\'{a}.

\end{estrofa}
