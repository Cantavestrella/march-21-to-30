\poema{(A 50) [23]}\\

\begin{rubrica}

Mossen auzias march \clauTancada

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Exant apart / lestil dels trobadors

\numero{2} Qui per escalf / trespassen veritat

\numero{3} E sostrahent / mon voler affectat

\numero{4} Per que nom trop / direl que trob en vos

\numero{5} Tot mon parlar / als qui no\textit{us} hauran vista

\numero{6} Res no valra / car fe noy donaran

\numero{7} Elos vehents / que dins vos no veuran

\numero{8} Encontramj / lur arma sera trista

\end{estrofa}



\begin{estrofa}

\numero{9} Lull del hom pech / no ha tan foscha vista

\numero{10} Que vostre cors / no jutge per gentil

\numero{11} Nol coneix tal / com lo qui es subtil

\numero{12} Hoc la color / mas no sab dela lista

\numero{13} Quant es del cors / menys de participar

\numero{14} Ab lespirit / coneix be lo grosser

\numero{15} Vostre color / yl tal pot be saber

\numero{16} Mas ia del gest / no pora be parlar

\end{estrofa}



\begin{estrofa}

\numero{17} [195r] Tots som grossers / en poder explicar

\numero{18} \c{C}o que mereix / vn bell cors e honest

\numero{19} Jouens gentils / bons sabents lan raquest

\numero{20} E fameiants / los couench endurar

\numero{21} Lo vostre seny / fa \c{c}o qualtre no basta

\numero{22} Que sab regir / la molta subtilea

\numero{23} En fer tot be / sadorm enuos parea

\numero{24} Verge no sou / perque deu ne volch casta

\end{estrofa}



\begin{estrofa}

\numero{25} Sol per auos / basta la bona pasta

\numero{26} Que deu retench / per fer singulars dones

\numero{27} Ffetas nhassats / molt sauies ebones

\numero{28} Mas compliment / dona teresal tasta

\numero{29} Hauent en si / tan gran coneximent

\numero{30} Que res nol fall / que tota nos conegua

\numero{31} Al hom deuot / sa bellesa ensegua

\numero{32} Past dentenents / es son entenjment

\end{estrofa}



\begin{estrofa}

\numero{33} Venecians / no han lo regiment

\numero{34} Tan paciffich / com vostre seny regeix

\numero{35} Subtilitats / quelentendre\textit{us} nodreix

\numero{36} E del cors bell / sens colpalmoujment

\numero{37} Tant gran delit / tot hom entenent ha

\numero{38} E occupat / se troban vos entendre

\numero{39} Que lo desig / del cors nos pot estendre

\numero{40} Aleig voler / ans com amort esta

\end{estrofa}
