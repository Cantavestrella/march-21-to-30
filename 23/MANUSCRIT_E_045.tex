\begin{estrofa}

\numero{1} Leixant apart / lestil dels trobados

\numero{2} qui per escalf / trespassen veritat

\numero{3} e sostraent mon voler affectat

\numero{4} per que nom torb / direl que trob en vos

\numero{5} Tot mon parlar / als qui nous hauran vista

\numero{6} res no valrra / car fe noy donaran

\numero{7} e los vehents / qui dins vos no veuran

\numero{8} en creure mi / llur arma sera trista

\end{estrofa}



\begin{estrofa}

\numero{12} hoc la color / mas no sab de la llista

\numero{14} ab lesperit / coneix be lo grosser

\end{estrofa}
