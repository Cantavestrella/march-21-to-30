\begin{estrofa}

\numero{1} [22r] Lexant apart lestil dels trobados

\numero{2} Qui solen dir mes dela veritat

\numero{3} E sostrahent mon voler affectat

\numero{4} Per que nom trop direl q\textit{ue} trop en vos

\numero{5} Tot mon parlar als qui nous haura\textit{n} vista

\numero{6} Res no valrra car fe noy donaran

\numero{7} E los vehents que dins vos no veuran

\numero{8} En contra mj lur arma sera trista

\end{estrofa}



\begin{estrofa}

\numero{9} Lull del hom pech no ha tan fosqua vista

\numero{10} Que vostre cos no jutje per gentil

\numero{11} Nol conex tal com lo qui es sobtil

\numero{12} Hoc la color mas no sab d\textit{e}la lista

\numero{13} Quant es del cos menys de parti\c{c}ipar

\numero{14} Ab lesperit conex be lo grocer

\numero{15} Vostre color el tal pot be saber

\numero{16} Mas ya del gest no pora be parlar.

\end{estrofa}



\begin{estrofa}

\numero{17} Tots son groces en poder explicar

\numero{18} \c{C}o que merex vn bell cos e honest

\numero{19} Jouens gentils bons sabents lan request

\numero{20} E famaiants los couench andurar

\numero{21} Lo vostre seny fa so qualtre no basta

\numero{22} Que sab regir la molta subtilea

\numero{23} En fer tot be sadorm en vos perea

\numero{24} Verge no sou per que deu ne volch casta

\end{estrofa}



\begin{estrofa}

\numero{25} Sol per a bos\footnote{\textit{Per} vos.} basta la bona pasta

\numero{26} Que deu retench per fer singulas dones

\numero{27} Ffetes na sats molt sauies e bones

\numero{28} Mas compliment dona Teresal tasta\footnote{Dona Teresal tasta
\textit{subratllats. La ratlla del subratlla s'est\'{e}n per part del marge
dret}.}

\numero{29} Hauent en si tan gran mereximent

\numero{30} Que res nol fall que tota nos conega

\numero{31} [22v] Al hom deuot sa bellessa len\c{c}ega

\numero{32} Past dentenents es son enteniment

\end{estrofa}



\begin{estrofa}
