\begin{estrofa}

\numero{1} Ant en amor / ma pensaconsentit

\numero{2} Que sens aquell / en als no puch entendre

\numero{3} Amj que plau / que dals no puch apendre

\numero{4} Tot altreffet / me corren gran despit

\numero{5} Los grans tresors / ne tot lonor del mon

\numero{6} Nom plau hauer / ab menys desser amat

\numero{7} Car sens a\c{c}o / nom de benuhirat

\numero{8} No es en mj / que desig me confon

\end{estrofa}



\begin{estrofa}

\numero{9} Pobre so donchs / molt mes que Job no fon

\numero{10} Puys es dit rich / \c{c}ell qui no ha desig

\numero{11} En passions / iom trob dins en lo mig

\numero{12} Si desigar / ab desesper lom fon

\numero{13} James amor / hac axj auinent

\numero{14} En be mostrar / sa famosa virtut

\numero{15} Com alt mu\textit{n}tar / mj quim trob abatut

\numero{16} Benuhirat / sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos extremps / amor es malmjre\textit{n}t

\numero{18} Per molt e poch / e lo mig se jaqueix

\numero{19} Bes mostra pech / puys contra si falleix

\numero{20} Car tot \c{c}o cau / questrem es son tinent

\numero{21} Bem marauell / si amor no decau

\numero{22} Puys quen extremps / vol questigason pes

\numero{23} Lo meu voler / es mes que tota res

\numero{24} El vostres menys quel ter\c{c} dun punt de dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot / si non es fet un gran frau

\numero{26} Trencant amor / de natura costums

\numero{27} Poch menys contrast / quen tenebres e lums

\numero{28} En mon voler / el demadona iau

\numero{29} O deu per que / amor es desagual

\numero{30} Que no consent / que vostre voler crescha

\numero{31} Perque lo meu / per negun temps parescha

\numero{32} Si be nom sent / quant me veuraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} [189r] Deu per bondat / vol ser tant cominal

\numero{34} Que no consent / vn cor dur e saluatge

\numero{35} Esser amat / agran desauentatge

\numero{36} Daquell qui es / en amor son cabal

\numero{37} Per\c{c}o no pens / que amor en mj dur

\numero{38} Car en amar / vos he pres tot estrem

\numero{39} E vostre cor / es damor axi sem

\numero{40} Quen mj pensar / no crech james atur

\end{estrofa}
