\poema{(F 21) [21]}\\

\begin{rubrica}

Altra

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Tant en amor mapensa consentit

\numero{2} Que sens aquell en als no puch entendre

\numero{3} A mj que plau que dals no puch apendre

\numero{4} Tot altraffer me corren gran despit

\numero{5} Los grans tresors ne tot lonor del mon

\numero{6} Nom plauhauer ab menys desser amat

\numero{7} Car sens aco nom de benahuyrat

\numero{8} No es en mj car desig me confon

\end{estrofa}



\begin{estrofa}

\numero{9} [21v] Pobre so donchs molt pus que Job no fon

\numero{10} Puys es dit rich cell quj no ha desig

\numero{11} En passions yom trob dins en lo mig

\numero{12} Si desijar ab desesper lom fon

\numero{13} James amor hac aci auinent

\numero{14} En be mostrar sa famosa uirtut

\numero{15} Com alt muntar mj qujm trob abatut

\numero{16} Benahuyrat sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos extrems amor es malmjrent

\numero{18} Per molt e poch e lo mig se iaqueix

\numero{19} Bes mostra pech puys contra si falleix

\numero{20} Car tot co cau questrem es son tinent

\numero{21} Bem marauell si amor no decau

\numero{22} Puys que estrems uol questiga son pes

\numero{23} Lo meu uoler es mes que tota res

\numero{24} El uostres menys quel terc dun punt de dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot sino mes fet gran frau

\numero{26} Trencant amor de natura costums

\numero{27} Poch menys contrast quentenebres e lums

\numero{28} En mon uoler yel demadona jau

\numero{29} O deu per que amor es desegual

\numero{30} Que no consent que uostre uoler cresqua

\numero{31} Per que lo meu per negun temps peresqua

\numero{32} Si be nom sent quantme ueuraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} [22r] Deu per bondat vol sser tan cominal

\numero{34} Que no conssent vn cor dur e saluatge

\numero{35} Esser amat agran desauantage

\numero{36} Daquell qui es en amor son cabal

\numero{37} Per co no pens que amor en mj dur

\numero{38} Car en amar uos hepres tot extrem

\numero{39} E uostre cor es damor axi ssem.

\numero{40} Quen mj penssar no crech james atur

\end{estrofa}



\begin{estrofa}

\numero{41} Plena de seny per lamor queus port jur

\numero{42} Que sim ue tart la uostra benuolenca

\numero{43} Present de tots fare de mj sentenca

\numero{44} Que sonara mentrel mon dels vius dur.

\end{estrofa}
