\begin{estrofa}

\numero{1} Tant en Amor / ma pens \sout{a} \interlin{ha} consentit

\numero{2} que sens aquell / en als no puch entendre

\numero{3} ami quemplau / que d'als no puch apendre

\numero{4} tot altre fer/ me correngran despit

\numero{5} Los grans tresors / ne tot l'h\interlin{o}nor del mon

\numero{6} nom plau haver / ab menys de ser amat

\numero{7} car sens a\c{c}o / nom de ben ahuirat\footnote{\textit{Escrit al marge esquerre
per una altra m\`{a}: }ben ahuirat / per benaven- / turat.}

\numero{8} \sout{no es encar desig molt me} \interlin{\c{c}o que no's car / tot mon desig} 
confon

\end{estrofa}



\begin{estrofa}

\numero{9} [20v] Pobre so donchs / molt mes que job no fon

\numero{10} puys es dit rich / cell qui no hadesig

\numero{11} enpassions / yomtrop\footnote{A continuaci\'{o}, lletra ratllada i
il$\cdot{}$legible.} dins en lo mig

\numero{12} sidesijar / ab desesper l'om fon

\numero{13} Jam es amor / hac axi avinent

\numero{14} en be mostrar / sa famosa virtut

\numero{15} com alt muntar / me tench trop abatut

\numero{16} ben ahuirat / sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Perdos extrems / amor es mal mirent

\numero{18} per molt he poch / \interlin{h}elo mig sa jaqueix

\numero{19} bes mostra pech / puis contra si falleix

\numero{20} car tot \c{c}o cau / questrem es son tinent

\numero{21} Bem marauell / siamor no decau

\numero{22} Puis qu'en estrems / vol questiga sonpes

\numero{23} lo meu voler / es mes que totares

\numero{24} yl vostres menys / quel ter\c{c} dun punct dedau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot / sino mes fet gran frau

\numero{26} trencant amor / de natura costums

\numero{27} poch menys contrast / que tenebres elums

\numero{28} en monvoler / yl de madona jau

\numero{29} Odeu perque / amor es desigual

\numero{30} que no consent / quevostre voler cresca

\numero{31} perque lo meu / per null temps no peresca

\numero{32} si be nom sent / quant mevendraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} Deus per bondat / vol ser tan cominal

\numero{34} que no consent / vncor dur e salvatje

\numero{35} esser amat / \interlin{h}a gran de sauantatje

\numero{36} daquell qui es / en amor son cabal

\numero{37} [21r] Per\c{c}o nompens / que amor en mi dur

\numero{38} car en amor / vos hepres tot estrem

\numero{39} e vostre cor / es damor axi sem

\numero{40} quen mi pensar / no creu james atur

\end{estrofa}
