\poema{(E 119) [21]}\\

\begin{rubrica}

Moss\textit{en} ausias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Tant en amor / ma pensa consentit

\numero{2} que sens aquell / en als no puch entendre

\numero{3} ami quem plau / que dals no pusch apendre\footnote{Mot manipulat.}

\numero{4} tot altre fer / me corren gran despit

\numero{5} Los grans tresors / ne tot lonor del mon

\numero{6} nom plau hauer / ab menys de ser amat

\numero{7} car sens ay\c{c}o / nom de ben ahuyrat

\numero{8} no es en mi / car desig me confon

\end{estrofa}



\begin{estrofa}

\numero{9} [183v] Pobre so donchs / molt pus que Job no fon

\numero{10} puis es dit rich / \c{c}ell qui no ha desig

\numero{11} en passions / yom trob dins en lo mig

\numero{12} si desijar / ab desesper hom fon

\numero{13} James amor / hac axi auinent

\numero{14} en ben mostrar / sa famosa virtut

\numero{15} com alt muntar / mi quim trob abatut

\numero{16} ben ahuyrat / sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos estrems / amor es mal mirent

\numero{18} per molt e poch / e lo mig se jaqueix

\numero{19} bes mostra pech / puis contra si falleix

\numero{20} car tot \c{c}o cau / questrem es sostinent

\numero{21} Bem marauell / si amor no decau

\numero{22} puis en estrems / vol questiga son pes

\numero{23} lo meu voler / es mes que tota res

\numero{24} yel vostre menys / quel ter\c{c} dun punt de Dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot / si nomes fet gran frau

\numero{26} trencant amor / de natura costums

\numero{27} poch menys contrast / quen tenebres e llums

\numero{28} est mon voler / yel de madona Jau

\numero{29} O deu per que / amor es desigual

\numero{30} que no consent / que vostre voler cresca

\numero{31} per que lo meu / per null temps no peresca

\numero{32} si be nom sent / quant me venraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} [184r] Deu per bondat / vol ser tan cominal

\numero{34} que no consent / vn cor dur e saluatge

\numero{35} esser amat / agran desauantatge

\numero{36} daquell qui es / en amor son cabal

\numero{37} Per \c{c}o no pens / que amor en mi dur

\numero{38} car en amar / vos he pres tot estrem

\numero{39} e vostre cor / es damor axi sem

\numero{40} quen mi pensar / no creu james atur

\end{estrofa}
