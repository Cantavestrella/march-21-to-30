\poema{(G2 91) [21]}\\

\begin{rubrica}

Moss\textit{en} auzias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Tant en amor / mapensaconssentit

\numero{2} que sens aquell / en als no pusch entendre

\numero{3} ami quem plau / quedals no puschapendre

\numero{4} tot altrafer / mecorren gran despit

\numero{5} Los grans tresors / netot lonor del mon

\numero{6} nom plau auer ab menys dess\textit{er} amat

\numero{7} car sens a\c{c}o / nom debenauyrat

\numero{8} no es en mi / car desig meconfon

\end{estrofa}



\begin{estrofa}

\numero{9} [131r] Pobreso donchs / molt pus q\textit{ue} Job no fon

\numero{10} puis es dit rich / cell qui nohadesig

\numero{11} enpassions yomtrob / dins en lo mig

\numero{12} si desijat / abdesesper lom fon

\numero{13} James amor / ach axi auinent

\numero{14} enbe mostrar safamosa virtut

\numero{15} com alt munar / miquim trob abatut

\numero{16} benauyrat / sus totscomplidame\textit{n}t

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos extrems / amor es mal mirent

\numero{18} per molt epoch / elomig ss\textit{er }jaqueix

\numero{19} bes mostrapech / puis contrasi falleix

\numero{20} car tot \c{c}o cau / que estrem essontinent

\numero{21} Bem marauell / Si amor no decau

\numero{22} puis en estrems / vol questigasonpes

\numero{23} lo meu voler es mes que tot arres

\numero{24} el vostres menys / quel ter\c{c} dun punt dedau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot / sino mes fet granfrau

\numero{26} trencant amor / de naturacostums

\numero{27} poch menys contrast / quen tenebres / ellums

\numero{28} en mon voler / yel demadonajau

\numero{29} O deu p\textit{er}que amor / es desegual

\numero{30} que no conssent / que n\textit{ost}re voler cresca

\numero{31} perque lo meu / per null t\textit{em}ps no peresca

\numero{32} Si be nom sent quant me veuraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} [131v] Deu per bondat / vol ser tancominal

\numero{34} que no consent / vn cor dur esaluatge

\numero{35} ess\textit{er} amat agran desauantatge

\numero{36} daquell qui es / enamor son cabal

\numero{37} Per\c{c}o no pens / queamor en mi dur

\numero{38} car enamor / vos he pres tot estrem

\numero{39} e vostre cor / es damor axissem

\numero{40} quen mi pensar / no creu james atur

\end{estrofa}
