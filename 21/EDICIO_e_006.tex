\poema{(e 6) [21]}\\

\begin{rubrica}

DE AMOR CANT. VI.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} T\textsc{a}nt en Amor / ma pens'ha consentit

\numero{2} que s\'{e}ns aquell / en als no puch ente\textit{n}dre

\numero{3} ami quem plau / que d'als no puch aprendre

\numero{4} tot altre fet / mi entre'n gran despit.

\numero{5} Los grans thresors / ne tot l'honor del mon

\numero{6} nom plau hauer / ab menys de ser amat

\numero{7} car sens a\c{c}o / nom de ben ahuirat

\numero{8} \c{c}o que no's car / tot mon desig confon.

\end{estrofa}



\begin{estrofa}

\numero{9} Pobre so donchs / molt mes que Iob no fon

\numero{10} Puix es dit rich / cell qui no a desig:

\numero{11} en passions / yom trob dins en lo mig

\numero{12} si desijar / ab desesper l'hom fon.

\numero{13} Iames Amor / fon aci auinent

\numero{14} en be mostrar / sa famosa virtut

\numero{15} com alt muntar / me, e ferme d'abatut

\numero{16} ben ahuirat / sus tots complidament.

\end{estrofa}



\begin{estrofa}

\numero{17} [6v] Per dos estrems / Amor es mal mirent

\numero{18} per molt e poch / e lo mig se jaqueix

\numero{19} be's mostra pech / puis contra si falleix

\numero{20} car tot \c{c}o cau / que'strem es son tinent.

\numero{21} Bem marauell / si Amor no decau

\numero{22} puix que'n estrems / vol que'stiga son pes

\numero{23} lo meu voler / es mes que tota res

\numero{24} yl vostre's menys / quel ter\c{c} d'vn pu\textit{n}t de dau.

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot / sino mes fet gran frau

\numero{26} trencant Amor / de natura'ls costums

\numero{27} poch menys contrast / que tenebres e lums

\numero{28} en mon voler / y'l de madona jau.

\numero{29} Deu per que / Amor es desigual

\numero{30} que no consent / que vostre voler cresca?

\numero{31} per que lo meu / per null temps no peresca

\numero{32} si be nom sent / quant me vendr'aquest mal.

\end{estrofa}



\begin{estrofa}

\numero{33} Deu per bondat / vol fer tan cominal

\numero{34} que no consent / vn cor dur e saluatje

\numero{35} e ser amat / a gran desauantatje

\numero{36} d'aquel qui es / en Amor son cabal.

\numero{37} Per \c{c}o nom pens / que amor en mi dur

\numero{38} car en amar / vos he pres tot estrem

\numero{39} el vostre cor / es d'amor axi sem

\numero{40} que'n mi pensar / no crech james atur.

\end{estrofa}
