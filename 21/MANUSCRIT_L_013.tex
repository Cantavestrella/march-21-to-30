\begin{estrofa}

\numero{1} Tant en amor / ma pensa consentit

\numero{2} Que sens aquell / en als no puch entendre

\numero{3} Amj que plau / que dals no puch apendre

\numero{4} Tot altreffer / me corren gran despit

\numero{5} Los grans tresors / ne tot lonor del mon

\numero{6} Nom plau hauer / ab menys desser amat

\numero{7} Car sens a\c{c}o / nom de benehuyrat

\numero{8} No es en mj / que desig me confon

\end{estrofa}



\begin{estrofa}

\numero{9} Pobre so donchs / molt mes que Job no fon

\numero{10} Puys es dit rich / cell quj no ha desig

\numero{11} En passions / jom trop dins en lo mig

\numero{12} Si desigar / ab desesper lom fon

\numero{13} James amor / hac axj auinent

\numero{14} En be mostrar / sa famosa virtut

\numero{15} Com alt muntar / mj qujm trop abatut

\numero{16} Benehuyrat / sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos extrems / amor es malmjrent

\numero{18} Per molt e poch / e lo mig se iaqueix

\numero{19} Bes mostra pech / puys contra si falleix

\numero{20} Car tot \c{c}o cau / questrem es son tinent

\numero{21} Bem marauell / si amor no decau

\numero{22} Puys quen extrems / vol questiga son pes

\numero{23} Lo meu voler / es mes que tota res

\numero{24} El vostre es menys / quel ter\c{c} dun punt de dau

\end{estrofa}



\begin{estrofa}

\numero{25} [146r] Durar no pot / si nom es fet gran frau

\numero{26} Trenchant amor / de natura costums

\numero{27} Poch menys contrast / quen tenebres e lums

\numero{28} En mon voler / yel dema dona iau

\numero{29} O deu per que / amor es desegual

\numero{30} Que no consent / que vostre voler crescha

\numero{31} Per que lo meu / per negun temps perescha

\numero{32} Si be nom sent / quant me venraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} Deu per bondat / vol ser tan cominal

\numero{34} Que no consent / vn cor dur e saluatge

\numero{35} Esser amat / agran desauantatge

\numero{36} Daquell quj es / en amor son cabal

\numero{37} Per \c{c}o no pens / que amor en mj dur

\numero{38} Car en amar / vos he pres tot estrem

\numero{39} E vostre cor / es damor axj sem

\numero{40} Quen mj pensar / no crech james atur

\end{estrofa}
