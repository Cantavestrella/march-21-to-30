\begin{estrofa}

\numero{1} \textsc{Ta}nt en amor / mapens'ha consentit

\numero{2} que sens aquell / en als no puch entendre,

\numero{3} ami quem plau / que d'als no puch apendre

\numero{4} tot altrefer /mi, entr'en gran despit.

\numero{5} Los grans tresors / ne tot l'honor del mon

\numero{6} [5v] nom plau hauer / ab menys de ser amat

\numero{7} car sens a\c{c}o /nom do, ben ahuirat

\numero{8} \c{c}o que no's car / tot mon desig confon.

\end{estrofa}



\begin{estrofa}

\numero{9} Pobre, so donchs / molt mes que Iob, no fon

\numero{10} puis es dit rich / cell qui no ha desig,

\numero{11} en passions / yom trob, dins en lo, mig

\numero{12} si desijar / ab, desesper, l'hom, fon.

\numero{13} james amor / fon, axi auinent

\numero{14} en be mostrar /sa famosa virtut

\numero{15} com al muntar / me, y ferme d'abatut,

\numero{16} ben ahuirat / sus tots, complidament.

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos extrems /amor, es mal mirent

\numero{18} per molt, e poch / e lo mig se jaqueix

\numero{19} bes mostra pech / puix contra si falleix

\numero{20} car tot \c{c}o, cau / qu'estrem es, son tinent.

\numero{21} b m\footnote{\textit{Hi ha un espai buit on hauria d'anar la vocal -e-} (bem).},
marauell /si, amor, no decau

\numero{22} puix qu'en estrems /vol que estiga son pes

\numero{23} lo meu voler / es mes que, tota, res

\numero{24} yl v\textit{ost}re's menys / quel ter\c{c} d'un punt de, dau.

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot / sino m'es fet gran frau

\numero{26} trencant, amor / de natura'ls costums

\numero{27} poch menys contrast /que tenebres, e, llums

\numero{28} en mon voler /y'l de, madona jau,

\numero{29} o Deu per que /amor, es desigual

\numero{30} que no consent / que vostre voler cresca

\numero{31} per que lo meu / per null temps no peresca,

\numero{32} si be nom sent / quant me vendr'aquest mal.

\end{estrofa}



\begin{estrofa}

\numero{33} [6r] Deus per bondat /vol ser tan cominal

\numero{34} queno consent /vn cor, dur, e seluatge,

\numero{35} esser, amat /a, gran desauantatge

\numero{36} d'aquel qui es /en amor son cabal.

\numero{37} per\c{c}o nom pens / que, amor, en mi dur

\numero{38} car, en amar /vos e pres tot, estrem

\numero{39} el vostre cor / es d'amor axi sem

\numero{40} que'n mi pensar / no crech james atur.

\end{estrofa}
