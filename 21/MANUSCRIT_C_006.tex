\begin{estrofa}

\numero{1} Tant'en amor / ma pensa ha consentit

\numero{2} que sens aquell / en als no puch entendre

\numero{3} ami quem plau / que d'als no puch apendre

\numero{4} tot altre fer / mientr' en gran despit.

\numero{5} los grans tesors / ne tot lhonor del mon

\numero{6} nom plau hauer / ab menys de ser amat

\numero{7} car sens a\c{c}o / nom do ben ahuirat

\numero{8} \c{c}o que no's car / tot mon desig confon

\end{estrofa}



\begin{estrofa}

\numero{9} Pobre so doncs / molt mes que Iob no fon

\numero{10} puis (*)\interlin{es} dit rich / cell qui no\textit{n} ha desig,

\numero{11} [13v] en passions / yom trob dins enlo mig

\numero{12} si desijar / ab desesper, l'hom fon.

\numero{13} Iames amor / fon, axi auinent

\numero{14} en be mostrar / sa famosa uirtut

\numero{15} com al muntar / me y ferme d'abattut

\numero{16} ben ahuyrat / sus tots cumplidament.

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos extrems / amor es mal mirent

\numero{18} per molt e poch / e lo mig se iaqueix

\numero{19} besmostra pech / puix contra si falleix

\numero{20} car tot \c{c}o cau / qu'extrem es son tinent

\numero{21} bem marauell / si amor no decau

\numero{22} puix qu'en extrems / uol qu'estiga son pes

\numero{23} lo meu uoler esmes / que tota res

\numero{24} y'l uostre's menys / quel ter\c{c} d'un punt de dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot / si nom es fet gran frau

\numero{26} trencant amor / de natura'l'scostums

\numero{27} poch menys contrast / que tenebres ellums

\numero{28} en mon uoler / y e'l de madona jau.

\numero{29} o deu perque / amor es desigual

\numero{30} que no consent / quel nostre uoler cresca

\numero{31} [14r] perque lomeu / per null temps no peresca

\numero{32} si be nom sent / quant me uendra aquestmal.

\end{estrofa}



\begin{estrofa}

\numero{33} Deu per bondat / uol ser tan cominal

\numero{34} que no consent / un cor dur e saluage

\numero{35} esser amat / a gran desauantage

\numero{36} d'aquel qui es / en amor son cabal

\numero{37} per \c{c}o nom pens / que'amor en mi dur

\numero{38} car en amar / uos e pres tot extrem

\numero{39} el nostre cor / es d'amor axi sem

\numero{40} quen'mi pensar / no crech iames atur.

\end{estrofa}
