\begin{estrofa}

\numero{1} [20v] Tant en amor pa penssa consentit

\numero{2} Que sens aq\textit{ue}ll en als no puch entendre

\numero{3} A mi que plau dals no poder apenrre

\numero{4} Tot altre fer me corren gran despit

\numero{5} Los (*) tresos e tot lonor d\textit{e}l mon

\numero{6} No (*) hauer ab menys desser amat

\numero{7} Car sens a\c{c}o nom de benuyrat

\numero{8} No es en mj car desitx me confon

\end{estrofa}



\begin{estrofa}

\numero{9} Pobre so donchs molt mes que job no fon

\numero{10} Puys es dit rich lo qui no ha desitx

\numero{11} En patrons jom trob en lo mjtx

\numero{12} Si desijar ab desesper lom fon

\numero{13} Y a mes amor ach axi auinent

\numero{14} En be mostrar sa famosa virtut

\numero{15} Com mu\textit{n}tar alt mi quim trobabatut

\numero{16} Benahuyrat sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos strems amor es mal mirent

\numero{18} Per molt epoch elo mjtx se jaquex

\numero{19} Bes mostra pech puyxs contra si falex

\numero{20} Car tot \c{c}o cau questrem es sentiment

\numero{21} Ben marauell si amor no decau

\numero{22} Puys quen strems vol questiga son pes

\numero{23} Lomeu voler es mes que tota res

\numero{24} El vostres menys que ter\c{c} dun punt d\textit{e} dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot si no mes fet gran frau

\numero{26} Trencant amor de natura costums

\numero{27} Poch menys constrast que tenebres e lums

\numero{28} E mon voler el de madona jau

\numero{29} O deu perque amor es desegual

\numero{30} Que no consent cresca vostre voler

\numero{31} Per que lo meu no perde son poder

\numero{32} Si be nom sent quant me veuraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} [21r] Deu per bondat vol ser tan comjnal

\numero{34} Que no consent vn cor dur e saluatie

\numero{35} Esser amat agran desauantatje

\numero{36} Daquell qui es en amor son cabal

\numero{37} Per\c{c}o no pens q\textit{ue} amor en mj dur

\numero{38} Car en amar vos he pres tot strem

\numero{39} E vostre cor es damor axi sem

\numero{40} Quen mj penssar no crech ja mes atur

\end{estrofa}
