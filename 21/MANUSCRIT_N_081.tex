\begin{estrofa}

\numero{1} \textparagraph\  Tant en amor mapensa consentit

\numero{2} Que sens aquell en als no puch entendre

\numero{3} A mj que plau que dals no puch apendre

\numero{4} Tot altre fer me corengran despit

\numero{5} Los grans tresors ne tot lonor del mon

\numero{6} Nom plau hauer abmeyns deser amat

\numero{7} Car sens aco nom de benauyrat

\numero{8} No es en mj car desig me confon

\end{estrofa}



\begin{estrofa}

\numero{9} [212] Pobre so donchs molt mes que job no fon

\numero{10} Pus es dit rich cell quj no ha desig

\numero{11} E passions yom trob dins enlo mig

\numero{12} Si desijar abdesesper lom fon

\numero{13} Ja mes amor hach axi aujnent

\numero{14} Enbe mostrar sa famosa virtut

\numero{15} Com alt muntar mj quem trob abatut

\numero{16} Beneuyrat sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos estrems amor es malmjrent

\numero{18} Per molt epoch elomig se jaquex

\numero{19} Bes mostra pech pus contra si fallex

\numero{20} Car tot co cau questrem es son tinent

\numero{21} Bem marauell fi amor no decau

\numero{22} Pus quen estrems uol questiga sonpes

\numero{23} Lo meu voler es mes que totares

\numero{24} El uostre meyns quel terch dun punt de dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot sino mes fet gran frau

\numero{26} Trencant amor denatura costums

\numero{27} Poch menys contrast quentenebres elums

\numero{28} En mon voler hi el demadama jau

\numero{29} Ho deu perque amor es desagual

\numero{30} Que no consent que nostre uoler cresca

\numero{31} Perque lomeu pernegun temps perescha

\numero{32} Sibe nom sent quant me vendraquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} [213] Deu per bondat uolch ser tan comjnal

\numero{34} Que no consent vn cor dur esaluatge

\numero{35} Esser amat agran desauentatge

\numero{36} Daquell quj es en amor son cabal

\numero{37} Perco no pens que amor enmj dur

\numero{38} Car en amar uos he pres tot strem

\numero{39} E uostre cor es damor axi sem

\numero{40} Quen mj pensar no crech james atur

\end{estrofa}
