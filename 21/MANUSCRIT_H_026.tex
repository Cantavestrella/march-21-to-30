\poema{(H 26) [21]}\\

\begin{rubrica}

Mos\textit{sen} auzias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Tant en amor / ma pen\c{c}a ha consentit

\numero{2} Que sens aquell en als no puch entendre

\numero{3} Amj que plau / que dals no pusca pendre

\numero{4} Tot altre ffer me corren gran despit

\numero{5} Los grans tresors / ne tot lonor del mon

\numero{6} Nom plau auer / ab menys desser amat

\numero{7} Car sens a\c{c}o / nom de benuyrat

\numero{8} No es amj car desig me confon

\end{estrofa}



\begin{estrofa}

\numero{9} [32r] Pobre so donchs / molt me\interlin{s} q\textit{ue} job no fon

\numero{10} Puys es dit rich / \c{c}ell q\textit{ui} no ha desig

\numero{11} En pasions / jom trob dins en lo mjg

\numero{12} Si desigar / ab desesper lom fon

\numero{13} James amor hach axi avjnent

\numero{14} En be mostrar / sa famossa virtut

\numero{15} Com al muntar / mj q\textit{ui}m trop abatut

\numero{16} Benuyrat / sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} Per dos estrems amor es mal mjrent

\numero{18} Per molt e poch e lo mjg se iaqueix

\numero{19} Bes mostra pech puys contra si faleix

\numero{20} Car tot \c{c}o cau questrem es son tinent

\numero{21} Bem merauell si amor no decau

\numero{22} Puys quen estrem vol questiga son pes

\numero{23} Lo meu voler es mes q\textit{ue} tota res

\numero{24} Lo vostres me\textit{n}ys quell ter\c{c} dun pu\textit{n}t de dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot sino mes fet gran frau

\numero{26} Trencant amor d\textit{e }natura costums

\numero{27} Pochs menys contrasts que tenebres e lums

\numero{28} En mon voler yell de madona jau

\numero{29} O deu p\textit{er} que amor es desiguall

\numero{30} Que no consent q\textit{ue} vostre voler cresca

\numero{31} Per q\textit{ue} lo meu per null temps no peresca

\numero{32} Si be nom sent quant me venra aquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} Deu \sout{ben} per bondat vol ser tan comjnall

\numero{34} Que no consent hun cor dur e saluatie

\numero{35} [32v] E ser amat agran desauentaje

\numero{36} Daquell quj es en amor son caball

\numero{37} Per \c{c}o no pens / que amor en mj dur

\numero{38} Car en amar / vos he pres tot estrem

\numero{39} E uostre cor es damor axi sem

\numero{40} Quen mj pensar no creu james atur

\end{estrofa}
