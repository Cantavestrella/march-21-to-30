\begin{estrofa}

\numero{1} Tant, en amor / mapens'ha consentit

\numero{2} que sens aquell / en als no puch ente\textit{n}dre,

\numero{3} ami quem plau / q\textit{ue}, d'als no pusch a pe\textit{n}dre

\numero{4} tot altre fer / mi, entre'n gran despit

\numero{5} los gra\textit{n}s tresors / ne tot l'honor del mon

\numero{6} nom plau hauer / ab menys de ser amat

\numero{7} car'sens a\c{c}o / nom do, ben ahuirat \cite{ref11}

\numero{8} \c{c}o que no's car / tot mon desig, co\textit{n}fon

\end{estrofa}



\begin{estrofa}

\numero{9} Pobre, so do\textit{n}chs / molt mes q\textit{ue}, job, no fon

\numero{10} puis es dit rich / cell qui no ha desig

\numero{11} en passions / yom tropdins enlo, mig

\numero{12} si desijar / ab, desesper, l'hom, fon

\numero{13} james amor / fon, axi auinent

\numero{14} enbe, mostrar / sa famosa virtut

\numero{15} com al mu\textit{n}tar / me, y ferme d'abatut

\numero{16} ben ahuirat / sus tots complidament

\end{estrofa}



\begin{estrofa}

\numero{17} [5v] Per dos extrems / amor, es mal mirent

\numero{18} per molt he poch / he lo mig se jaqueix

\numero{19} bes mostra pech / puix contra si falleix

\numero{20} car tot \c{c}o cau, que'strem, es, son tinent

\numero{21} bem, marauell / si, amor, no decau

\numero{22} puix qu'en estrems / vol q\textit{ue} estiga son pes

\numero{23} lo meu voler / es mes que, tota, res

\numero{24} yl v\textit{ost}re's menys / q\textit{ue}l ter\c{c} dun punt de, dau

\end{estrofa}



\begin{estrofa}

\numero{25} Durar no pot sino mes fet gran frau

\numero{26} trencant, amor / de natura costums

\numero{27} poch me\textit{n}ys co\textit{n}trast / q\textit{ue} tenebres, he, llums

\numero{28} en mon voler / y'l de, madona, jau

\numero{29} ho deu per que / amor, es desigual

\numero{30} que no consent / q\textit{ue} vostre voler cresca

\numero{31} per que lo meu / per null te\textit{m}ps no peresca

\numero{32} si be nom sent / qua\textit{n}t me vendr'aquest mal

\end{estrofa}



\begin{estrofa}

\numero{33} Deus per bondat / vol ser tan cominal

\numero{34} que no co\textit{n}sent / vn cor, dur, he seluatge

\numero{35} esser, amat / ha, gran desauantatge

\numero{36} daquel qui es / e'n amor son cabal

\numero{37} per\c{c}o nom pe\textit{n}s / que, amor, en mi dur

\numero{38} car, en amar / vos he pres tot, estrem

\numero{39} el vostre cor / es d'amor axi sem \cite{ref12}

\numero{40} que'n mi pensar / no crech, james atur

\end{estrofa}
