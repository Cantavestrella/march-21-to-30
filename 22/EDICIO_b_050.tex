\begin{estrofa}

\numero{1} C\textsc{a}llen aquells / qui d'amor, han parlat

\numero{2} he dels passats / delint tots, lurs scrits

\numero{3} en mi pensant / meten los, en oblit

\numero{4} en mon esguart / nengu, es'namorat

\numero{5} [44v] car, pas desig / sens speran\c{c}a hauer

\numero{6} tal passio / james home, sostench

\numero{7} per, als damnats / nostre deu, la retench

\numero{8} sols per aquells / qui moren sens, esper

\end{estrofa}



\begin{estrofa}

\numero{9} Puis mon afany / es entre tots, primer

\numero{10} amor fa tort / com se rete, mon dret

\numero{11} all\footnote{\textit{Per }als.} folls, he pechs / non dona fam, ne set

\numero{12} ne fret, ne calt / tots han, complit, mester

\numero{13} car, no'n noblex / tota merce, son alt

\numero{14} merce hauent / si james, la demana

\numero{15} yo d'amor visch / he desperan\c{c}a, vana

\numero{16} si pens hauer / raho, de fret ne, calt

\end{estrofa}



\begin{estrofa}

\numero{17} No so empes / ne temprat, per desalt

\numero{18} de res que'n vos / los meus vlls, haje\textit{n} vist

\numero{19} l'enteniment / per lo vostre's, conquist

\numero{20} en gran raho / es causat, aquest, alt

\numero{21} vos no veheu / ma pura'ntencio

\numero{22} car, desamor / vos en fosqueix, la vista

\numero{23} per \c{c}o romanch / abla mia'rma, trista

\numero{24} com no mirau / la mia'fectio

\end{estrofa}



\begin{estrofa}

\numero{25} Del pare sanct / nom cal, hauer perdo

\numero{26} car, mon peccat / es amar follament

\numero{27} de man l'ami / car per mon, sentiment

\numero{28} he fet d'amor / catiua, ma raho

\numero{29} sia content / amor del cor, sens pus

\numero{30} he la  raho / seruira, son offici

\numero{31} mon sentiment / ques veu, prim, sia ni\c{c}i

\numero{32} que no sesguart / ab qui tinga lull, clus

\end{estrofa}



\begin{estrofa}

\numero{33} [45r] Fent be a molts / ala mort, me acus

\numero{34} car, sere spil / dels leals, amadors

\numero{35} prenint, remey / a totes lurs, dolors

\numero{36} car, en vers mi / tota dolor, es jus

\numero{37} ay tant com he / pus alt, leuada pensa

\numero{38} e netament / hay, en amor, entes

\numero{39} seny femeni / fora si, ma empes

\numero{40} fent asi tort / e a mi, gran offensa

\end{estrofa}
