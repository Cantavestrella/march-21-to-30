\poema{(E 26) [22]}\\

\begin{rubrica}

Moss\textit{en }ausias march

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Callen aquells / qui damor han parlat

\numero{2} e dels passats / deliu tots llur scrits

\numero{3} en mi pensant / meteu los en oblits

\numero{4} en mon sguart / degus enamorat

\numero{5} Car pas desig / sens speran\c{c}a ver

\numero{6} tal passio / james home sostench

\numero{7} perals damnats / nostre Deu la retench

\numero{8} sol peraquells / qui moren sens sper

\end{estrofa}



\begin{estrofa}

\numero{9} [37r] Puis mon affany / es entre tots primer

\numero{10} amor faytort / com se rete mon dret

\numero{11} als fats e pechs / no dona fam ne set

\numero{12} ne fret ne calt / tots han complit mester

\numero{13} Car non es hu / no trobe tot son alt

\numero{14} mer\c{c}e hauent / si james la demana

\numero{15} e yo non trob / vna dona tan sana

\numero{16} quem fart lo seny / em fa\c{c}a ser malalt\footnote{Els dos darrers versos
d'aquesta estrofa -com tamb\'{e} els quatre primers de l'estrofa seg\"{u}ent-
semblen manipulats i alguns mots afegits per una altra m\`{a}.}

\end{estrofa}



\begin{estrofa}

\numero{17} No suy empes / ne temptat per desalt

\numero{18} de res qua vos / haja donat natura

\numero{19} de quant pot dar / vos no passats fretura

\numero{20} mas vostre cor / vers amor no es calt

\numero{21} E no vehets / ma purantencio

\numero{22} car desamor / vos enfosqueix la vista

\numero{23} per\c{c}o romanch / ab la mi arma trista

\numero{24} com no mirats / la miafectio

\end{estrofa}



\begin{estrofa}

\numero{25} Del pare sanct / nom cal hauer perdo

\numero{26} car mon peccat / es amar follament

\numero{27} deman la mi / com tan acordament

\numero{28} ha catiuats / sentiments e raho

\numero{29} Sia content / amor del cos sens pus

\numero{30} e la raho / seruira son offici

\numero{31} mon sentiment / quis veu prim sia nici

\numero{32} que nosesguart / ab qui tinga lull clus

\end{estrofa}



\begin{estrofa}

\numero{33} [37v] Fent be amolts / ala mort me acus

\numero{34} car serespill / de lleals amadors

\numero{35} prenints remey / a totes llurs dolors

\numero{36} car en vers mi / tota dolor es jus

\numero{37} Aytant com he / pus alt lleuada pensa

\numero{38} e netament / ay en amor entes

\numero{39} seny femenil / fora si ma empes

\numero{40} fent asi tort / e ami gran offensa

\end{estrofa}

\begin{estrofa}

\numero{41} Plena de seny / yo vull e deu dispensa

\numero{42} que per amor / yo fenesca mos jorns

\numero{43} mas sin escap / per null temps daretorns

\numero{44} per dona quen / ver amor se deffensa
