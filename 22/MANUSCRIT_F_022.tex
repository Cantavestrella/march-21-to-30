\poema{(F 22) [22]}\\

\begin{rubrica}

Altra

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} Callen aquells que damor han parlat

\numero{2} E dels passats deliu tots lurs escrits

\numero{3} En mj penssant meteu los en oblits

\numero{4} En mon esguart degus enamorat

\numero{5} Car pas desig sensesperancauer

\numero{6} Tal passio james home sostench

\numero{7} Per als dampnats nostre deu la retench

\numero{8} Sol per aquells quj moren sens esper

\end{estrofa}



\begin{estrofa}

\numero{9} [22v] Puys mon afany es entre tots primer

\numero{10} Amor fa tort com se rete mon dret

\numero{11} Als fals e pechs non dona fam ne set

\numero{12} Ne fret ne calt tots han complit mester

\numero{13} Car non es hu no trobe tot son alt

\numero{14} Merce hauent si iames la demana

\numero{15} E yo damor visch desperanca vana

\numero{16} Sjm pens hauer raho desserne calt

\end{estrofa}



\begin{estrofa}

\numero{17} No so empes ne temptat per desalt

\numero{18} Deres quen uos los me\textit{us} hulls hagen vist

\numero{19} Lenteniment per los uostres conquist

\numero{20} En gran raho es causat aquest alt

\numero{21} Vos no ueheu mapurantencio

\numero{22} Car desamor uos enfosqueix la uista

\numero{23} Per co romanch ab lamiarma trista

\numero{24} Com no meriu la mja ffeccio

\end{estrofa}



\begin{estrofa}

\numero{25} Del pare sant nom cal hauer perdo

\numero{26} Car mon peccat es amar follament

\numero{27} Deman la mj cab mon consentiment

\numero{28} He fet damor catiua ma raho

\numero{29} Sia content amors del cors sens pus

\numero{30} E la raho seruira son offici

\numero{31} Mon sentiment qujs ueu prim sia njci

\numero{32} Que no sesguart ab quj tinga lull clus

\end{estrofa}



\begin{estrofa}

\numero{33} [23r] Ffent be amolts a la mort me acus

\numero{34} Car sere spill de leals amadors

\numero{35} Prenjnts remey atotes lurs dolors

\numero{36} Car enuers mj tota dolor es jus

\numero{37} Aytant com he pus alt leuada pensa

\numero{38} E netament hay en amor entes

\numero{39} Seny femenil fora si ma empes

\numero{40} Ffent assi tort e amj gran offensa

\end{estrofa}



\begin{estrofa}

\numero{41} Plena de seny yo vull e deu dispensa

\numero{42} Que per amor yo fenesqua mos jorns

\numero{43} Mas sim escap per null temps dare torns

\numero{44} Per dona que ueramor se deffensa

\end{estrofa}
