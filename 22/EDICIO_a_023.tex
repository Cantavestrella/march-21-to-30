\begin{estrofa}

\numero{1} P\textsc{u}ix mon affany / es entre tots primer

\numero{2} Amor fa tort / com se rete mon dret

\numero{3} Als falts e pechs / no dona fam ni set

\numero{4} Ne fret ne calt / tots han complit mester

\numero{5} Car no es hu / que no trop tot son alt

\numero{6} Merce hauent / si james la demana

\numero{7} E yo damor / visch desperan\c{c}a vana

\numero{8} Sim pens hauer / raho de serne calt

\end{estrofa}



\begin{estrofa}

\numero{9} \textparagraph\  No son en pes / nitentat per desalt

\numero{10} De res quen vos / los meus vlls hajen vist

\numero{11} Lenteniment / per lo vostre es conquist

\numero{12} En gran raho / es causat aquest alt

\numero{13} Mas vos no veu / ma pura intencio

\numero{14} Car desamor / vos enfosquex la vista

\numero{15} Per so roman / ab la mi arma trista

\numero{16} Com no mirau / la mia affeccio.

\end{estrofa}



\begin{estrofa}

\numero{17} [41r] \textparagraph\  Damor nom cal hauer james perdo

\numero{18} De mon peccat / ques amar follament

\numero{19} Dauant mi cau / tot mon consentiment

\numero{20} Fet he damor / catiua ma raho:

\numero{21} Sia content / del cos sens voler pus

\numero{22} E la  raho / seruira son offici

\numero{23} Mon sentiment / quis veu prim sia nici

\numero{24} Que no sesguart / al qui tinga lull clus

\end{estrofa}



\begin{estrofa}

\numero{25} \textparagraph\  Fent be a molts / a la mort me acus

\numero{26} Car sere espill / de lleals amadors

\numero{27} Prenint remey / a totes llurs dolors

\numero{28} Car en vers mi / tota dolor es jus

\numero{29} Aytant com he / pus alt llevada pensa

\numero{30} E netament / hay en amor entes

\numero{31} Seny feminil / fora si ma empes

\numero{32} Fent a si tort / e ami gran ofensa

\end{estrofa}
