\begin{estrofa}

\numero{1} C\textsc{a}llen aquells / qui d'amor han parlat

\numero{2} e dels passats / delits tots lurs scrits

\numero{3} en mi pensant / meten los en oblits

\numero{4} en mon esguart / nengu es'namorat.

\numero{5} car pas desig / sens speran\c{c}a hauer

\numero{6} [48r] \footnote{En realitat \'{e}s el full 47r.}tal passio / james home sastench

\numero{7} per als dampnats / nostre Deu la retench

\numero{8} sols per aquells / qui moren sens'sper.

\end{estrofa}



\begin{estrofa}

\numero{9} Puis mon afany / es entre tots primer.

\numero{10} amor fa tort / com se rete mon dret

\numero{11} als folls e pechs / non dona fam, ne set,

\numero{12} ne fret, ne calt / tots han complit mester.

\numero{13} car no'n nobleix / tota merce son alt,

\numero{14} merce hauent / si james la demana

\numero{15} yo d'amor visch / e desperan\c{c}a vana

\numero{16} si pens hauer / raho, de fret ne calt.

\end{estrofa}



\begin{estrofa}

\numero{17} No so empes / ne temprat, per desalt,

\numero{18} de res que'n vos / los meus vlls hajen vist:

\numero{19} l'enteniment / per lo vostr's conquist

\numero{20} engran raho / es causat aquest alt.

\numero{21} vos no veheu / ma pur'intencio,

\numero{22} car, desamor / vos en fosqueix la vista

\numero{23} per \c{c}o romanch / ab la mia'rma trista

\numero{24} com no mirau / la mia'ffectio.

\end{estrofa}



\begin{estrofa}

\numero{25} Del pare sanct / nom cal hauer perdo

\numero{26} car mon peccat / es amar follament

\numero{27} de man l'ami / car per mon sentiment

\numero{28} e fet d'amor / catiua, ma raho.

\numero{29} sia content / amo\footnote{\textit{Per} amor.} del cor sens pus.

\numero{30} e la raho / seruira son offici

\numero{31} mon sentiment / ques veu prim, sia nici:

\numero{32} que no sesguart / ab qui tinga lull clus.

\end{estrofa}



\begin{estrofa}

\numero{33} [48v] \footnote{\textit{En realitat \'{e}s el full 47v}.}Fent be a molts / a la
mort me acus.

\numero{34} car sere'spil / dels leals amodors\footnote{\textit{Per} amadors.}

\numero{35} prenint remey / a totes lurs dolors

\numero{36} car enuers mi / tota dolor es jus.

\numero{37} ay tant com e / pus alt leuada pensa

\numero{38} e netament / hay en amor entes

\numero{39} seny femeni / fora si ma empes

\numero{40} fentasi tort / e ami gran offensa.

\end{estrofa}

\numero{41} Plena de seny / yo vull y Deu dispensa

\numero{42} que per amar / yo fenesca mos jorns.

\numero{43} mas si escap / per null temps dare torns

\numero{44} per dona que'n / ver amor se defensa:
