\begin{estrofa}

\numero{1} [68r] Callen aquells / qui d'amor han parlat

\numero{2} e dels passats / delint tots lurs escrits

\numero{3} en mi pensant / metenlos en oblits

\numero{4} en mon esguart / nengu es'namorat.

\numero{5} car pass desig / sin esperan\c{c}a hauer

\numero{6} tal passio / james home sastench

\numero{7} per als dampnats / nostre Deu la retench

\numero{8} sols per aquells / qui moren sens'sper.

\end{estrofa}



\begin{estrofa}

\numero{9} Puis mon afany / entre tots primer

\numero{10} amor fa tort / com serete mondret

\numero{11} alls folls e pechs / non dona fam eset

\numero{12} ne fret ne calt / tots han complit mester

\numero{13} car no'n nobleix / tota mer\c{c}e son alt

\numero{14} mer\c{c}e hauent / si iames la demana

\numero{15} yo d'amor uisch / e de speran\c{c}a uana

\numero{16} si pens hauer raho / de fret ne calt.

\end{estrofa}



\begin{estrofa}

\numero{17} No so empes / ne temprat per desalt

\numero{18} de res que'n uos / los meus ulls hajen uist

\numero{19} l'enteniment / per lo uostre's conquist

\numero{20} en gran raho / es causat aquest alt.

\numero{21} [68v] uos no ueheu / ma pur intentio

\numero{22} car desamor / uos enfosquex la uista

\numero{23} per\c{c}o romanch / ab la mi'arma trista

\numero{24} com no mirau / la mia'ffectio.

\end{estrofa}



\begin{estrofa}

\numero{25} Del pare sanct / nom cal hauer perdo

\numero{26} car mon peccat / es amar follament

\numero{27} deman l'ami / car per mon sentime\textit{n}t

\numero{28} e fet d'amor / catiua ma raho.

\numero{29} sia content / amor del cor sens pus

\numero{30} e la raho / seuira son offici

\numero{31} mon sentiment / ques ueu prim sia nicj

\numero{32} que no sesguart / ab qui tinga lull clus.

\end{estrofa}



\begin{estrofa}

\numero{33} Fent be a molts / ala mort me acus

\numero{34} car sere spill / dels leals amadors

\numero{35} prenint remey / a totes lurs dolors

\numero{36} car en uers mi / tota dolor es jus

\numero{37} aytant com e / pus alt leuada pensa

\numero{38} e netament / hai en amor entes

\numero{39} seny femeni / fora si ma empes

\numero{40} fent asi tort / y ami gran offensa.

\end{estrofa}
