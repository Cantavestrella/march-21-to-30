\begin{estrofa}

\numero{1} \textparagraph\  Callen aquells quj damor han parlat

\numero{2} E dels passats deliu tots los scrits

\numero{3} En mj pensant meteulos enoblits

\numero{4} En mon esguart degus enamorat

\numero{5} Car pas desig sens esperancauer

\numero{6} Tal passio gemes home sostench

\numero{7} Per als dampnats nostre deu laretench

\numero{8} Sol per aquells quj moren sens esper

\end{estrofa}



\begin{estrofa}

\numero{9} [44] Puys mon affay es entre tots primer

\numero{10} Amor fa tort com se rete mon dret

\numero{11} Als fals he pechs non dona fam ne set

\numero{12} Ne fret ne calt tots an complit mester

\numero{13} Car noes hu no trobe tot son alt

\numero{14} Merce hauent si ja mes la demana

\numero{15} E yo damor visch desperanca vana

\numero{16} Sim pens auer raho desser ne calt

\end{estrofa}



\begin{estrofa}

\numero{17} No so enpes ne temptat per desalt

\numero{18} Deres quen vos los me\textit{us} hulls agen vist

\numero{19} Lenteniment per lo vostres conquist

\numero{20} En gran raho es causat aquest alt

\numero{21} Vos no v\interlin{e}heu ma purantencio

\numero{22} Car desamor uos enfosquex lavista

\numero{23} Perco romanch enlamiarma trista

\numero{24} Com no meriu lamia fectio

\end{estrofa}



\begin{estrofa}

\numero{25} Del pare sant nom cal hauer perdo

\numero{26} Car mon peccat es amar follament

\numero{27} Demanlamj quab mon consentiment

\numero{28} He fet damor catiua ma raho

\numero{29} Sia content amor del cors sens p\textit{us}

\numero{30} E la raho seruira son officj

\numero{31} Mon sentiment quis veu \sout{s} prim sia nici

\numero{32} Que no sesguart abquj tingua lull clus

\end{estrofa}



\begin{estrofa}

\numero{33} [45] Fent be amolts ala mort me acus

\numero{34} Car sere spill de leals amados

\numero{35} Prenints remey atotes lurs dolos

\numero{36} Car enves mj tota dolor es gu\textit{us}

\numero{37} Aytant comhe p\textit{us} alt leuada pensa

\numero{38} Enetament he enamor entes

\numero{39} Seny femenil fora sima enpes

\numero{40} Ffent asi tort camj gran offensa

\end{estrofa}
