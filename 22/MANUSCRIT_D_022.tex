\begin{estrofa}

\numero{1} Callen aquells / qui damor hanparlat

\numero{2} edels passats / deliu tots los lurs scrits

\numero{3} en mi pensant / meteulos enoblits

\numero{4} en mon esguart / nengu es namorat

\numero{5} Carpas design / sens esperan\c{c}a hauer

\numero{6} tal passio james / home sostench

\numero{7} per als damnats / no stre deu laretench

\numero{8} sol per aquells / qui moren sens esper

\end{estrofa}



\begin{estrofa}

\numero{9} Puis mon afany / es entre tots primer

\numero{10} amor fatort / com serete mondret

\numero{11} als folls epechs / nondona fam ne set

\numero{12} ne fret ne calt / tots han cumplit mester

\numero{13} Car nos vn noble / tot merce son alt

\numero{14} merce hauent / si ja mes lademana

\numero{15} yo damor visch / desesperan\c{c}a vana

\numero{16} sipens hauer raho / de ser ne calt

\end{estrofa}



\begin{estrofa}

\numero{17} [21v] No so emp es ne temprat / perdesalt

\numero{18} de\sout{u} res quen vos / los meus vlls haien vist

\numero{19} lenteniment / per lo vostres conquist

\numero{20} engran raho / es causat aquest alt

\numero{21} Vos no veheu / mapurantencio

\numero{22} car desamor / vos en fo squeix lavista

\numero{23} per \c{c}o romanc / abla mia arma trista

\numero{24} comno mirau / la mia ffectio

\end{estrofa}



\begin{estrofa}

\numero{25} Del pare sanct / nom cal hauer perdo

\numero{26} car mon pecat / es amar follament

\numero{27} demanlamj / ca\interlin{r} p\textit{er} mon sentiment

\numero{28} he fet damor / catiua maraho

\numero{29} Siacontent amor / del cor sens pus

\numero{30} elaraho / seruira son offici

\numero{31} mon sentiment quis veuprim sia nici

\numero{32} que no sesguart / ab quitinga llul clus

\end{estrofa}



\begin{estrofa}

\numero{33} Ffentbe amolts / a la mort me acus

\numero{34} car serespill / dels leals amadors

\numero{35} prenint remey / atotes lurs dolors

\numero{36} car en vers mi / tota dolor es jus

\numero{37} Aytant com he / pus alt leuadapensa

\numero{38} enetament / ay en amor entes

\numero{39} seny feminil / forasi ma empres

\numero{40} fent asitort / e ami gran ofensa

\end{estrofa}



\begin{estrofa}

\numero{41} Plena de seny / yo vull y deu despensa

\numero{42} que per amar / yo fenesca mos jorns

\numero{43} mas sin escap / per null temps dare torns

\numero{44} perdona / que\textit{n}ver amor sedefensa

\end{estrofa}
