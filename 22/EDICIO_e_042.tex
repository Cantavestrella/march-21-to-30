\poema{(e 42) [22]}\\

\begin{rubrica}

DE AMOR CANT. XLII.

\end{rubrica}

\end{estrofa}


\begin{estrofa}

\numero{1} [46r] C\textsc{a}llen aquells / qui d'Amor an parl\'{a}t

\numero{2} e dels passats / delint tots llurs escrits

\numero{3} en mi pensant / meten los en oblits

\numero{4} en mon esguart / nengu es namor\'{a}t.

\numero{5} Car pas desig / sens esperan\c{c}a hau\'{e}r

\numero{6} tal passio / james home sost\'{e}nch

\numero{7} per als damnats / nostre Deu la ret\'{e}nch

\numero{8} sols per aquells / qui moren sens esp\'{e}r.

\end{estrofa}



\begin{estrofa}

\numero{9} Puis mon affany / es entre tots primer

\numero{10} Amor fa tort / com se rete mon dret

\numero{11} als fals e pechs / no dona fam ne set

\numero{12} ne fret ne calt / tots han complit mester.

\numero{13} Car no'n noblex / tota merce son alt

\numero{14} merce hauent / si james la demana

\numero{15} yo d'Amor visch / e de'speran\c{c}a vana

\numero{16} si pens hauer / raho de fret ne calt.

\end{estrofa}



\begin{estrofa}

\numero{17} No so enpes / ne temptat per desalt

\numero{18} de res qu'en vos / los meus vlls hajen vist

\numero{19} l'enteniment / per lo vostre's conquist

\numero{20} en gran raho / es causat aquest alt.

\numero{21} Vos no veheu / ma pura intencio

\numero{22} car desamor / vos enfosqueix la vista

\numero{23} per\c{c}o romanch / ab la mi'arma trista

\numero{24} com no mirau / la mia affectio.

\end{estrofa}



\begin{estrofa}

\numero{25} [46v] Del pare sant / no cal hauer perdo

\numero{26} car mon peccat / es amar follam\'{e}nt

\numero{27} deman l'ami / car per mon sentim\'{e}nt

\numero{28} e fet d'Amor / catiua ma raho.

\numero{29} Sia content / Amor del cos sens pus

\numero{30} e la raho / seruira son off\^{\i}ci

\numero{31} mon sentiment / ques veu prim sia nici

\numero{32} que no se'sguart / ab qui tinga l'ull clus.

\end{estrofa}



\begin{estrofa}

\numero{33} Fent be a molts / a la mort me ac\'{u}s

\numero{34} car sere'spill / dels altres amadors

\numero{35} prenint remey / a totes llurs dolors

\numero{36} car en vers mi / tota dolor es jus.

\numero{37} Ay tant com he / pus alt leuada pensa

\numero{38} e netament / ay en Amor ent\'{e}s

\numero{39} seny feminil / fora si m'a enp\'{e}s

\numero{40} fent a si tort / e ami gran offensa.

\end{estrofa}

\begin{estrofa}

\numero{41} Plena de seny / yo vull y Deu dispensa

\numero{42} que per amar / yo fenesca mos jorns

\numero{43} mas sin escap / per null temps dare torns

\numero{44} per dona qu'en / ver amor se defensa.

\end{estrofa}
